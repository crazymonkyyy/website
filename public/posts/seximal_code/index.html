struct sexy(T:int){
	T i; alias i this;
	dchar digit(T j){
		enum dchar zero='☐';
		enum dchar one='⚀';
		assert(j>=0 && j<=6);
		if(j==0){ return zero;}
		return one+(j-1);
	}
	dstring toString(){
		auto j=i;
		dstring out_;
		bool neg= j<0;
		if(j<0){j=-j;}
		while(j!=0){
			out_=digit(j%6)~out_;
			j=j/6;
		}
		if(out_.length==0){return [digit(0)];}
		return out_;
	}
}
